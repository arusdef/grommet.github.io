FormField

A field in a web form.

Options

error {string|node}
Validation errors.
help {string|node}
Helpful text.
htmlFor {string}
Id of the input element that the label should be associated with.
label {string|node}
Label for the field.

Example

Text input

<Form>
  <FormField label="Item 1" htmlFor="item1">
    <input id="item1" type="text" />
  </FormField>
</Form>

Text input with errors

error text
<Form>
  <FormField label="Item 1" htmlFor="item1" error="error text">
    <input id="item1" type="text" />
  </FormField>
</Form>

Checkbox

<Form>
  <FormField label="">
    <CheckBox id="item2" label="Item 2" />
  </FormField>
</Form>

Range

<Form>
  <FormField label="item 6" htmlFor="item6">
    <input id="item6" type="range" min="0" max="80" defaultValue="40" />
  </FormField>
</Form>

RadioButton with help

help text
<Form>
  <FormField label="item 3" help="help text">
    <RadioButton id="item3-1" label="choice 1" name="choice" />
    <RadioButton id="item3-2" label="choice 2" name="choice" />
  </FormField>
</Form>

NumberInput

<Form>
  <FormField label="item 4" htmlFor="item4">
    <NumberInput id="item4" name="number" />
  </FormField>
</Form>

SearchInput

<Form>
  <FormField label="item 5" htmlFor="item5">
    <SearchInput id="item5" defaultValue="This is a really long search value that keeps going and going. Wow it is really long!" />
  </FormField>
</Form>

DateTime

<FormField label="item 7" htmlFor="item7">
  <DateTime id="item7" onChange={...} />
</FormField>
